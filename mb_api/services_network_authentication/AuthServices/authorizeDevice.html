Future<void> authorizeDevice(
  BuildContext context, {
  required Function onSuccess,
}) async {
  final postEndpoint = PostRequest(endpoint: kDeviceRegister);
  final controller = ConfirmController();
  final deviceId = await DeviceInfoServices.getId();
  final deviceName = await DeviceInfoServices.getName();

  final showMaximumDeviceDialog = () {
    DialogsServices().showCustom(
      context,
      title: "Authorize Device Failed",
      content: MAX_DEVICE_ERR,
      mainButtonTitle: "Okay",
      mainButtonOnPressed: () => Navigator.of(context).pop(),
    );
  };

  await Navigator.of(context).push(_nav.createRoute(
    AuthScreen(
      title: "Authorize Device",
      controller: controller,
      otpType: eOTPType.Both,
      validationEndpoint: PostRequest(
          endpoint: kRequestDeviceRegister,
          body: jsonEncode({
            "deviceId": deviceId.encodeBase64(),
            "deviceName": deviceName.encodeBase64(),
          })),
      onValidationError: (response) {
        // maximum device error code
        if (response!.statusCode == 403) showMaximumDeviceDialog();
      },
      confirmEndpoint: postEndpoint,
      onValidated: (otpRes) {
        postEndpoint.body = jsonEncode({
          "deviceId": deviceId.encodeBase64(),
          "deviceName": deviceName.encodeBase64(),
          "otp": otpRes.base64Otp,
          "otpReference": otpRes.otpReference,
        });
      },
      confirmWidget: SuccessScreen(
        header: "New Device Found",
        subHeader: "Allow this device to access your account?",
        textColor: Theme.of(context).primaryColor,
        backgroundColor: Colors.white,
        button: "Allow",
        popAllOnFinished: false,
        onFinish: () => controller.Confirm(),
      ),
      onSuccess: (res) => onSuccess(),
    ),
  ));
}
