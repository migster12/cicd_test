  BuildContext context, {
  required Function onSuccess,
}) async {
  //setSpinner(true);
  final hash = EncryptionAlgorithm.hash(password.text);
  if (password.text.length < 8 || hash.characters.elementAt(hash.characters.length - 3) != "0" && (hash.endsWith("00") || hash.endsWith("0"))) {
    _dialogs.showCustom(
      context,
      title: "Oops",
      content: "The password you created is weak. Are you sure you'd like to continue with this password?",
      enableSecondaryButton: true,
      mainButtonTitle: "Continue",
      secondaryButtonTitle: "Cancel",
      mainButtonOnPressed: () {
        Navigator.of(context).pop();
        onSuccess();

      },
    );
  } else {
    onSuccess();
  }
