}) async {
  setSpinner(true);

  final Uri uri = Uri.https(kServerAddress, kPostBillsPaymentURI);

  final requestBody = jsonEncode(<String, dynamic>{
    "sourceAccount": selectedAccount!.number,

  return await http

    uri,
    headers: kHTTPHeaders,
    body: requestBody,
  )
      .then((http.Response response) async {
    if (response.body.isNotEmpty && response.statusCode == 200) {
      setSpinner(false);

      final String _referenceNo = json.decode(response.body)["document"];

      Navigator.of(context).push(_nav.createRoute(
        BillsPaymentSuccessScreen(
          serviceCharge: fee,
          imageUrl: imageUrl,
          referenceNo: _referenceNo,
        ),

      setSpinner(false);

      final String message = json.decode(response.body)["message"];
      SnackBarServices.showError(context, message);
      throw "An unexpected error occurred.\nError Code: ${response.statusCode}";
