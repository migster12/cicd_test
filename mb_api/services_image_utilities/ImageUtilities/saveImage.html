static Future<void> saveImage(BuildContext context, {required Uint8List? image}) async {
  bool denied = false;
  if (Platform.isAndroid) {
    bool granted = false;
    var androidInfo = await DeviceInfoPlugin().androidInfo;
    if (androidInfo.version.sdkInt >= 33)
      granted = await Permission.photos.request().isGranted;
    else
      granted = await Permission.storage.request().isGranted;

    if (granted) {
      // Either the permission was already granted before or the user just granted it.'
      _saveImage(context, image!);
    } else {
      var status = await Permission.storage.status;
      if (status == PermissionStatus.denied || status == PermissionStatus.permanentlyDenied) denied = true;
    }
  } else {
    var isSuccess = await _saveImage(context, image!);
    if (!isSuccess) {
      denied = true;
    }
  }
  if (denied) {
    DialogsServices().showCustom(
      context,
      content: 'Unable to continue, please configure permission settings.',
      title: 'Check App Settings',
      mainButtonOnPressed: () {
        openAppSettings();
        Navigator.of(context).pop();
      },
      mainButtonTitle: 'Okay',
    );
  }
}
