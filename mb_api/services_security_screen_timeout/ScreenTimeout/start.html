static start() {
  _lastActiveTime = DateTime.now();
  _idleTimer = Timer.periodic(Duration(minutes: 1), (timer) => checkIdle(timer: timer));
  _isScreenActive = true;

  var sessionTime = CyberONEServices.appConfiguration!.sessionTokenValidityMinutes;
  if (sessionTime <= 0) {
    print("Screen Timeout disabled");
    return;
  }
  _isTimeout = false;
  _timer = Timer(Duration(minutes: sessionTime.toInt()), _onTimeout);

  //Timer(Duration(seconds: 10), _onTimeout);
}
