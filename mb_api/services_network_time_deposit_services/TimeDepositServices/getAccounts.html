Future<List<TimeDepositAccountResponse>> getAccounts(BuildContext context) async {
  final request = PostRequest(endpoint: kLoadAccountsTDURI);
  try {
    return await ApiUtilities.httpRequest(
      request,
      method: "post",
    ).then((response) async {
      var responseList = json.decode(response!)["payload"] as List;
      accounts = responseList.map((json) => TimeDepositAccountResponse.fromJson(json)).toList();
      notifyListeners();
      if (accounts!.isNotEmpty) {
        updateSelectedAccount(accounts![0]);
      }
      return accounts!;
    });
  } on SocketException {
    return _handleSocketExceptionError(context);
  } on Exception catch (e) {
    SnackBarServices.showError(context, e.getMessage);
  }
  return List.empty();
}
