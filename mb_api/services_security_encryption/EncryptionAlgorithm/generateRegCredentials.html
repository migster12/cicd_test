static Future<RegCredentials> generateRegCredentials({
  required String email,
  required String mobile,
  required String username,
  required String password,
  required String deviceId,
  required String deviceName,
}) async {
  final diff = computeDifficulty(username);
  final hUsername = hash(username, difficulty: diff);
  final cfu = hUsername.encodeBase64();
  final column = BigInt.parse(hUsername, radix: 16) % BigInt.from(149);
  final cfc = column.toInt().toString().encodeBase64();
  final passServer = hash(password, difficulty: diff);
  final passArgon = await EncryptionAlgorithm.argonHash(passServer);
  final cfp = passArgon.encodeBase64();
  return RegCredentials(
    email: email.encodeBase64(),
    mobile: mobile.encodeBase64(),
    hColumn: cfc,
    hUsername: cfu,
    hPassword: cfp,
    deviceId: deviceId.encodeBase64(),
    deviceName: deviceName.encodeBase64(),
  );
}
