static Future<String> getId() async {
  if (_retrievedId != "") return _retrievedId;

  var deviceInfo = DeviceInfoPlugin();
  if (Platform.isIOS) {
    String? val;
    try {
      val = await FlutterKeychain.get(key: IOS_KEY.md5Hash());
      if (val != null && val.length > 0) {
        _retrievedId = val;
        return val;
      }
    } catch (e) {}

    // import 'dart:io'
    var iosDeviceInfo = await deviceInfo.iosInfo;
    _retrievedId = iosDeviceInfo.identifierForVendor + iosDeviceInfo.model; // unique ID on iOS
    await FlutterKeychain.put(key: IOS_KEY.md5Hash(), value: _retrievedId);
  } else if (Platform.isAndroid) {
    var androidDeviceInfo = await deviceInfo.androidInfo;
    _retrievedId = androidDeviceInfo.androidId + androidDeviceInfo.model; // unique ID on Android
  }
  return _retrievedId;
}
