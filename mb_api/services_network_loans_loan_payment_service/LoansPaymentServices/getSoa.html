Future<Soa?> getSoa({required String accountNumber, required double amount}) async {
  final _uri = kComputeLoanPaymentsURI;

  return await ApiUtilities.request(_uri, method: "post", body: jsonEncode(<String, dynamic>{"accountNumber": accountNumber, "paymentAmount": amount}))
      .then((http.Response response) {
    print("Response Body: ${response.body}");
    if (response.statusCode == 200 && response.body.isNotEmpty) {
      final Soa _soaData = Soa.fromJson(jsonDecode(response.body)["payload"]);
      return _soaData;
    } else {
      SnackBarServices.showError(navigatorKey.currentContext!, UNEXPECTED_ERR);
      return null;
      //throw "Error Code: ${response.statusCode}";
    }
  });
}
